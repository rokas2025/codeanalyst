async function importKeywordsFromWebsiteAnalysis(projectId) {

    SELECT data FROM project_data
    WHERE project_id = $1 AND module = 'website-analyst' 
      AND data_type = 'seo-keywords'

async function executeWorkflow(workflowId, userInput, userId) {
  const workflow = workflowTemplates[workflowId];
  const context = { user_input: userInput };
  
  for (let i = 0; i < workflow.steps.length; i++) {
    const step = workflow.steps[i];
    
    // Execute step
    const result = await executeModule(step.module, step.action, 
      interpolateVariables(step.input, context)
    );
    
    // Store result in context
    context[step.output] = result;
    
    // Checkpoint: wait for user approval
    if (workflow.checkpoints.includes(i)) {
      await waitForUserApproval(userId, workflowId, i, result);
    }
  }
  
  return context;
}

// backend/src/middleware/auth.js
export async function authenticateToken(req, res, next) {
  const token = req.headers.authorization?.replace('Bearer ', '');
  
  if (!token) {
    return res.status(401).json({ error: 'No token provided' });
  }
  
  try {
    // Verify JWT
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    
    // Fetch user from DB
    const user = await db.query(
      'SELECT * FROM users WHERE id = $1 AND active = true',
      [decoded.userId]
    );
    
    if (user.rows.length === 0) {
      return res.status(401).json({ error: 'User not found' });
    }
    
    // Attach user to request
    req.user = user.rows[0];
    
    // Check subscription access
    if (!hasModuleAccess(req.user.subscription_tier, req.path)) {
      return res.status(403).json({ 
        error: 'Upgrade to access this feature',
        requiredTier: getRequiredTier(req.path)
      });
    }
    
    next();
  } catch (error) {
    return res.status(401).json({ error: 'Invalid token' });
  }
}

// backend/src/middleware/apiKeyAuth.js
export async function authenticateAPIKey(req, res, next) {
  const apiKey = req.headers['x-api-key'];
  
  if (!apiKey) {
    return res.status(401).json({ error: 'API key required' });
  }
  
  // Verify API key
  const keyData = await db.query(`
    SELECT ak.*, u.id as user_id, u.subscription_tier
    FROM api_keys ak
    JOIN users u ON ak.user_id = u.id
    WHERE ak.key = $1 AND ak.active = true
  `, [apiKey]);
  
  if (keyData.rows.length === 0) {
    return res.status(401).json({ error: 'Invalid API key' });
  }
  
  // Check rate limit
  const rateLimit = await checkRateLimit(apiKey, keyData.rows[0].subscription_tier);
  if (rateLimit.exceeded) {
    return res.status(429).json({ 
      error: 'Rate limit exceeded',
      limit: rateLimit.limit,
      resetAt: rateLimit.resetAt
    });
  }
  
  req.apiKey = keyData.rows[0];
  req.user = { id: keyData.rows[0].user_id };
  next();
}

// backend/src/services/aiService.js
import OpenAI from 'openai';
import Anthropic from '@anthropic-ai/sdk';

class AIService {
  constructor() {
    this.openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
    this.anthropic = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });
  }
  
  async chat(messages, options = {}) {
    const model = options.model || 'gpt-4-turbo';
    const maxTokens = options.maxTokens || 4096;
    
    try {
      let response;
      
      if (model.startsWith('gpt')) {
        response = await this.openai.chat.completions.create({
          model: model,
          messages: messages,
          max_tokens: maxTokens,
          temperature: options.temperature || 0.7
        });
        
        // Track usage
        await this.trackUsage(options.userId, model, response.usage);
        
        return {
          content: response.choices[0].message.content,
          model: model,
          tokensUsed: response.usage.total_tokens
        };
        
      } else if (model.startsWith('claude')) {
        response = await this.anthropic.messages.create({
          model: model,
          max_tokens: maxTokens,
          messages: messages
        });
        
        await this.trackUsage(options.userId, model, {
          prompt_tokens: response.usage.input_tokens,
          completion_tokens: response.usage.output_tokens,
          total_tokens: response.usage.input_tokens + response.usage.output_tokens
        });
        
        return {
          content: response.content[0].text,
          model: model,
          tokensUsed: response.usage.input_tokens + response.usage.output_tokens
        };
      }
      
    } catch (error) {
      // Fallback to GPT-4 if primary model fails
      if (model === 'gpt-5-preview') {
        logger.warn('GPT-5 failed, falling back to GPT-4');
        return await this.chat(messages, { ...options, model: 'gpt-4-turbo' });
      }
      throw error;
    }
  }
  
  async trackUsage(userId, model, usage) {
    await db.query(`
      INSERT INTO ai_usage_logs (user_id, model, prompt_tokens, completion_tokens, total_tokens, timestamp)
      VALUES ($1, $2, $3, $4, $5, NOW())
    `, [userId, model, usage.prompt_tokens, usage.completion_tokens, usage.total_tokens]);
  }
}

export default new AIService();

  const moduleResults = {
    codeAnalysis: project.modules_data.find(m => m.module === 'code-analyst'),
    websiteAnalysis: project.modules_data.find(m => m.module === 'website-analyst'),
    autoProgrammer: project.modules_data.find(m => m.module === 'auto-programmer'),
    contentCreator: project.modules_data.find(m => m.module === 'content-creator')
  };
  
  // Generate HTML
  const html = generateHTMLReport({
    projectName: project.name,
    date: new Date().toISOString(),
    modules: moduleResults
  });
  
  if (format === 'html') {
    return { html: html, filename: `${project.name}_report.html` };
  }
  
  // Generate PDF
  const browser = await puppeteer.launch({ headless: true });
  const page = await browser.newPage();
  await page.setContent(html);
  
  const pdf = await page.pdf({
    format: 'A4',
    printBackground: true,
    margin: { top: '20mm', bottom: '20mm', left: '15mm', right: '15mm' }
  });
  
  await browser.close();
  
  return { pdf: pdf, filename: `${project.name}_report.pdf` };
}

// backend/src/services/projectGenerator.js
import OpenAI from 'openai';

async function generateProject(prompt, language) {
  const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
  
  // Step 1: Plan architecture
  const architecturePlan = await openai.chat.completions.create({
    model: 'gpt-4-turbo',
    messages: [{
      role: 'system',
      content: 'You are a software architect. Plan project structure.'
    }, {
      role: 'user',
      content: `Plan file structure for: "${prompt}" in ${language}`
    }],
    response_format: { type: 'json_object' }
  });
  
  const plan = JSON.parse(architecturePlan.choices[0].message.content);
  // Example plan: { files: ['src/index.js', 'src/routes/users.js', 'package.json'] }
  
  // Step 2: Generate each file
  const files = await Promise.all(
    plan.files.map(async (filePath) => {
      const fileContent = await generateFileContent(filePath, prompt, plan);
      return { path: filePath, content: fileContent };
    })
  );
  
  // Step 3: Generate README
  const readme = await generateReadme(prompt, plan, files);
  files.push({ path: 'README.md', content: readme });
  
  return {
    files: files,
    structure: plan,
    entryPoint: plan.entryPoint || 'src/index.js'
  };
}

// backend/src/services/codeValidator.js
import { Linter } from 'eslint';
import * as parser from '@typescript-eslint/parser';

async function validateGeneratedCode(files) {
  const linter = new Linter();
  const results = [];
  
  for (const file of files) {
    if (file.path.match(/\.(js|ts)$/)) {
      // Syntax validation
      try {
        parser.parse(file.content, { 
          ecmaVersion: 2022, 
          sourceType: 'module' 
        });
      } catch (syntaxError) {
        results.push({
          file: file.path,
          type: 'error',
          message: `Syntax error: ${syntaxError.message}`,
          line: syntaxError.lineNumber
        });
        continue;
      }
      
      // ESLint validation
      const lintResult = linter.verify(file.content, {
        parserOptions: { ecmaVersion: 2022, sourceType: 'module' },
        rules: {
          'no-unused-vars': 'warn',
          'no-console': 'warn',
          'no-undef': 'error'
        }
      });
      
      results.push(...lintResult.map(r => ({
        file: file.path,
        type: r.severity === 2 ? 'error' : 'warning',
        message: r.message,
        line: r.line
      })));
    }
  }
  
  return {
    errors: results.filter(r => r.type === 'error').length,
    warnings: results.filter(r => r.type === 'warning').length,
    issues: results
  };
}

import archiver from 'archiver';

async function exportAsZip(files, projectName) {
  const archive = archiver('zip', { zlib: { level: 9 } });
  const outputPath = `/tmp/${projectName}.zip`;
  const output = fs.createWriteStream(outputPath);
  
  archive.pipe(output);
  
  // Add each file to ZIP
  for (const file of files) {
    archive.append(file.content, { name: file.path });
  }
  
  await archive.finalize();
  
  return outputPath; // Download URL
}

// backend/src/services/codeRefinement.js
async function refineCode(previousFiles, refinementPrompt, context) {
  const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
  
  // Build context with previous code
  const previousCode = previousFiles.map(f =>
    `File: ${f.path}\n${f.content}`
  ).join('\n\n');
  
  const response = await openai.chat.completions.create({
    model: 'gpt-4-turbo',
    messages: [
      {
        role: 'system',
        content: 'You are refining existing code based on user feedback.'
      },
      {
        role: 'user',
        content: `Previous code:\n${previousCode}\n\nUser feedback: "${refinementPrompt}"\n\nModify the code accordingly. Return ONLY the files that changed.`
      }
    ],
    response_format: { type: 'json_object' }
  });
  
  const modifiedFiles = JSON.parse(response.choices[0].message.content);
  
  // Merge with unchanged files
  const updatedFiles = previousFiles.map(f => {
    const modified = modifiedFiles.files.find(mf => mf.path === f.path);
    return modified || f;
  });
  
  return {
    files: updatedFiles,
    changedFiles: modifiedFiles.files.map(f => f.path),
    diff: generateDiff(previousFiles, updatedFiles)
  };
}
